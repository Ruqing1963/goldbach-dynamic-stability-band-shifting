\begin{abstract}
We study the evolution of the Goldbach conductor distribution as the even integer
$N$ varies continuously from $2^{10}$ to $2^{11}$.  The conductor proxy of the
Goldbach--Frey Jacobian $J_{N,p}$ depends on $N$ through the static conduit factor
$\rad_{\mathrm{odd}}(N/2)$, which fluctuates erratically---producing step-to-step
jumps in Chen's ratio $\rho$ exceeding $\pm 3$ units.  Despite this apparent
instability, we discover three rigidity phenomena:
(1)~The \emph{Band Shifting Law}: the mean Goldbach ratio satisfies
$\langle\rho\rangle_{\mathrm{GB}} = 1.994\,\xi + 3.424$ where
$\xi = 2\log(\rad_{\mathrm{odd}}(M))/\log N$, with $R^2 = 0.9974$---the ``wild''
jumps are $99.7\%$ deterministic.
(2)~\emph{Bandwidth constancy}: the width of the Goldbach $\rho$-band is
$1.09 \pm 0.20$ across all 513 values of $N$, invariant under band shifting.
(3)~\emph{Non-emptiness under jump}: the Goldbach locus $\mathscr{G}(N)$ is
non-empty for all 513 even integers in $[1024, 2048]$, despite conductor
discontinuities.  These results demonstrate that the Goldbach phenomenon exhibits
\emph{dynamic stability}: the locus adapts to arbitrarily large conductor
perturbations by reorganising its prime-pair content while preserving its
structural width.
\end{abstract}

\section{Introduction}

In the companion papers~\cite{Chen2026GM,Chen2026GMII,Chen2026AV}, we studied the
conductor geometry of the Goldbach--Frey family
\begin{equation}\label{eq:curve}
  C_{N,p} : y^2 = x(x^2 - p^2)(x^2 - q^2), \qquad q = N - p,
\end{equation}
through the normalised conductor proxy (Chen's ratio)
\begin{equation}\label{eq:rho}
  \rho(N, p) = \frac{\log \mathcal{N}_{\mathrm{proxy}}}{\log N},
  \qquad
  \mathcal{N}_{\mathrm{proxy}} =
    \bigl(\rad_{\mathrm{odd}}(p)\cdot\rad_{\mathrm{odd}}(q)\cdot
          \rad_{\mathrm{odd}}(M)^2\bigr)^2,
\end{equation}
where $M = N/2$.  The factor $\rad_{\mathrm{odd}}(M)$ is the \emph{static conduit}:
it depends only on $N$, not on the decomposition parameter $p$.

The preceding analysis~\cite{Chen2026GMII} studied $\rho$ for \emph{fixed} $N$;
the algebraic vacuum paper~\cite{Chen2026AV} specialised to $N = 2^k$ where the
static conduit vanishes.  A natural question is: \emph{how does the conductor
distribution evolve as $N$ varies?}

This question is motivated by a na\"ive proof strategy: if one could construct a
``propagation operator'' $\mathcal{P}: \mathcal{M}_N \to \mathcal{M}_{N+2}$ that
preserves the Goldbach locus, then non-emptiness at $N = 2^k$ (established
in~\cite{Chen2026AV}) would imply non-emptiness for all $N$.  We show that this
strategy \emph{fails} in its original form: the static conduit
$\rad_{\mathrm{odd}}(M)$ is discontinuous in $N$, so no smooth propagation
operator exists.  However, the failure is itself informative---it reveals a
far more interesting phenomenon that we call \emph{dynamic stability}.\footnote{Scripts
and data:
\url{https://github.com/Ruqing1963/goldbach-dynamic-stability-band-shifting}.}

\section{The Discontinuity of the Static Conduit}\label{sec:discontinuity}

\subsection{The conduit jump}

\begin{proposition}[Conduit discontinuity]\label{prop:jump}
  The function $N \mapsto \rad_{\mathrm{odd}}(N/2)$ is discontinuous at every
  $N$ that transitions between different odd-radical classes.  In particular:
  \begin{enumerate}
  \item At $N = 2^k$: $\rad_{\mathrm{odd}}(2^{k-1}) = 1$.
  \item At $N = 2^k + 2$: $\rad_{\mathrm{odd}}(2^{k-1} + 1)$ is typically
    a product of several odd primes, often $\gg 1$.
  \end{enumerate}
  The jump $\Delta\!\rad = \rad_{\mathrm{odd}}((2^k+2)/2) - 1$ can be arbitrarily
  large.
\end{proposition}

\begin{proof}
  Immediate from the arithmetic of $\rad_{\mathrm{odd}}$.  For example,
  $\rad_{\mathrm{odd}}(512) = 1$ but $\rad_{\mathrm{odd}}(513) = \rad_{\mathrm{odd}}(3^3 \cdot 19) = 57$.
\end{proof}

\begin{remark}[Failure of smooth propagation]\label{rem:failure}
  Any putative propagation operator $\mathcal{P}: \mathcal{M}_N \to \mathcal{M}_{N+2}$
  must contend with the fact that the conductor environment changes discontinuously.
  The transition $N = 1024 \to 1026$ shifts $\rho_{\min}$ by $+2.48$---not a perturbation
  but a topological reconfiguration.  Consequently, inductive arguments of the form
  ``$\mathscr{G}(N) \neq \emptyset \Rightarrow \mathscr{G}(N+2) \neq \emptyset$''
  cannot work through conductor continuity alone.
\end{remark}


\subsection{Decomposition of Chen's ratio}

The key to understanding the $N$-dependence is to separate $\rho$ into two additive
components.

\begin{definition}[Conduit--boundary decomposition]\label{def:decomp}
  For $N = 2M$ and a Goldbach pair $(p, q)$ with $\rad_{\mathrm{odd}}(p) = p$,
  $\rad_{\mathrm{odd}}(q) = q$:
  \begin{equation}\label{eq:decomp}
    \rho(N, p) = \underbrace{\frac{2\log(\rad_{\mathrm{odd}}(M)^2)}{\log N}}_{\displaystyle
      \xi(N) \;\text{(conduit contribution)}}
    \;+\; \underbrace{\frac{2\log(p \cdot q)}{\log N}}_{\displaystyle
      \beta(N,p) \;\text{(boundary contribution)}}.
  \end{equation}
  The conduit contribution $\xi(N)$ depends only on $N$; the boundary contribution
  $\beta(N,p)$ depends on the specific Goldbach pair.
\end{definition}

\section{The Band Shifting Law}\label{sec:shifting}

\subsection{Empirical discovery}

We scan all 513 even integers $N \in [1024, 2048]$ and compute, for each $N$,
the mean Chen's ratio $\langle\rho\rangle_{\mathrm{GB}}$ over all Goldbach
pairs.\footnote{The complete dataset (513 rows) is available in the companion
repository.}

\begin{theorem}[Band Shifting Law]\label{thm:BSL}
  Over $N \in [1024, 2048]$, the mean Goldbach ratio satisfies the linear relation
  \begin{equation}\label{eq:BSL}
    \langle\rho\rangle_{\mathrm{GB}}(N) \;=\; \alpha\,\xi(N) + \gamma,
  \end{equation}
  with fitted parameters
  \begin{equation}\label{eq:params}
    \alpha = 1.994 \pm 0.005, \qquad \gamma = 3.424 \pm 0.006,
    \qquad R^2 = 0.9974.
  \end{equation}
\end{theorem}

\begin{remark}[Algebraic interpretation of $\alpha \approx 2$]\label{rem:alpha}
  The slope $\alpha \approx 2$ is not a numerical coincidence.  In the conductor
  proxy~\eqref{eq:rho}, the static conduit enters as $\rad_{\mathrm{odd}}(M)^2$
  (squared), contributing $4\log(\rad_{\mathrm{odd}}(M))/\log N = 2\xi$ to $\rho$.
  The fitted slope $\alpha = 1.994$ is thus a direct algebraic consequence of the
  discriminant formula $\Delta = 2^{12}\,p^6\,q^6\,(M-p)^4\,M^4$.
  The slight deviation from $2$ is attributable to the averaging over $\beta(N,p)$
  across different Goldbach pairs, which introduces a weak $N$-dependent correction.
\end{remark}

\begin{remark}[Interpretation of $\gamma \approx 3.4$]\label{rem:gamma}
  The intercept $\gamma = 3.424$ is the mean boundary contribution:
  $\gamma = \langle\beta\rangle = \langle 2\log(pq)/\log N \rangle$ averaged
  over Goldbach pairs.  At $N = 2^k$ (where $\xi = 0$), the Band Shifting Law
  predicts $\langle\rho\rangle \approx 3.4$, consistent with the values
  $3.29$--$3.55$ observed in~\cite{Chen2026AV}.
\end{remark}

\begin{remark}[Intercept drift and the corrected BSL]\label{rem:drift}
  A sensitivity test across wider $N$-ranges reveals that $\gamma$ is not a
  universal constant but drifts slowly upward (Table~\ref{tab:sensitivity}).
  The slope $\alpha$, by contrast, converges toward the algebraic value~$2$.

  \begin{table}[H]
  \centering
  \begin{tabular}{lrrrrr}
  \toprule
  Range & $n$ & $\alpha$ & $\gamma$ & $R^2_{\mathrm{mean}}$ & $R^2_{\mathrm{min}}$ \\
  \midrule
  $[2^{10}, 2^{11}]$ & 511   & 1.994 & 3.424 & 0.9974 & 0.920 \\
  $[2^{11}, 2^{12}]$ & 1\,023 & 1.999 & 3.466 & 0.9985 & 0.921 \\
  $[2^{12}, 2^{13}]$ & 2\,047 & 1.999 & 3.512 & 0.9990 & 0.915 \\
  $[2^{13}, 2^{14}]$ & 4\,095 & 2.000 & 3.548 & 0.9994 & 0.913 \\
  \bottomrule
  \end{tabular}
  \caption{Band Shifting Law parameters across different $N$-ranges.
  The slope $\alpha \to 2$ (algebraically exact).  The intercept $\gamma$ drifts:
  $3.42 \to 3.47 \to 3.51 \to 3.55$.}
  \label{tab:sensitivity}
  \end{table}

  The drift has a clean analytic explanation.
  For a ``typical'' Goldbach pair $(p, q)$ with $p$ near $N/2$, the boundary
  contribution is $\beta = 2\log(pq)/\log N \approx 4 - c/\log N$ where
  $c > 0$ depends on the prime distribution.  Thus $\gamma(N) \to 4$ from below
  as $N \to \infty$.  The \emph{corrected} Band Shifting Law is
  \begin{equation}\label{eq:BSL_corrected}
    \langle\rho\rangle_{\mathrm{GB}}(N) = 2\,\xi(N) + 4 - \frac{c}{\log N}
    + O\!\left(\frac{1}{\log^2 N}\right),
  \end{equation}
  where the slope is now exact and the intercept is an explicit function of $N$.
  Within any fixed dyadic interval $[2^k, 2^{k+1}]$, the local approximation
  $\gamma \approx \mathrm{const}$ remains excellent ($R^2 \geq 0.997$), and
  improves as $k$ increases.
\end{remark}

\section{The Jump--Conduit Correlation}\label{sec:correlation}

If the Band Shifting Law is exact, then step-to-step jumps in $\rho$ should be
entirely explained by changes in the conduit contribution.  We test this.

\begin{proposition}[Jump correlation]\label{prop:corr}
  Define $\Delta\rho_{\min}(N) = \rho_{\min}(N+2) - \rho_{\min}(N)$ and
  $\Delta\xi(N) = \xi(N+2) - \xi(N)$.  Over $N \in [1024, 2046]$ (512 steps):
  \begin{equation}\label{eq:corr}
    \mathrm{Corr}(\Delta\xi,\, \Delta\rho_{\min}) = 0.981.
  \end{equation}
\end{proposition}

This correlation of $0.981$ transforms the interpretation of the conductor jumps.
What appears to be ``chaotic'' behaviour in $\rho$ is in fact $98\%$ deterministic:
the jumps are controlled by the arithmetic of $\rad_{\mathrm{odd}}(M)$.

\begin{observation}[The residual $2\%$]\label{obs:residual}
  The residual $1 - 0.981^2 \approx 3.8\%$ of variance in $\Delta\rho_{\min}$
  that is \emph{not} explained by $\Delta\xi$ arises from the boundary contribution
  $\beta$: when $N$ changes, the identity of the ground-state Goldbach pair
  $(p_0, q_0)$ can change discretely, producing a residual fluctuation in
  $\rho_{\min}$ beyond what the conduit shift predicts.  However, this residual
  has zero mean and standard deviation $0.035$---it is arithmetic noise, not
  structural instability.
\end{observation}

\section{Non-emptiness Under Jump}\label{sec:nonempty}

\subsection{The computational result}

\begin{observation}[Universal non-emptiness]\label{obs:nonempty}
  For every even $N \in [1024, 2048]$ (513 values), the Goldbach locus
  $\mathscr{G}(N) = \{p : p \text{ and } N - p \text{ both prime}\}$ is non-empty.
  The minimum number of Goldbach pairs is $G(N) \geq 11$ (achieved at
  $N = 1024$, $2048$ and their $2^k$ neighbours), while the maximum exceeds $80$.
\end{observation}

\begin{remark}[Bandwidth vs.\ prime density: why non-emptiness is robust]\label{rem:density}
  A natural concern is whether there exists an extreme $N$ whose conduit shift
  pushes the Goldbach band into a region devoid of prime pairs.  We show this is
  implausible.  The $\rho$-band of width $\mathrm{BW} \approx 1.09$ maps to a
  range of summand values $\Delta p \sim \mathrm{BW} \cdot N\log N / 4$.
  By the prime number theorem, this range contains approximately
  $\Delta p / \log N \sim \mathrm{BW} \cdot N / 4$ primes.  Concretely:

  \begin{center}
  \begin{tabular}{rrrr}
  \toprule
  $N$-range & $\mathrm{BW_{GB}}$ & Primes in band & Min.\ $G(N)$ \\
  \midrule
  $[1024,\, 2048]$  & 1.09 & $\sim 420$ & 16 \\
  $[4096,\, 8192]$  & 1.20 & $\sim 1840$ & 45 \\
  $[8192,\, 16384]$ & 1.25 & $\sim 3830$ & 75 \\
  \bottomrule
  \end{tabular}
  \end{center}

  The supply of candidate primes within the $\rho$-band grows as $O(N)$, while
  the bandwidth itself is $O(1)$ in $\rho$-space.  For the Goldbach locus to
  become empty, \emph{every} prime $p$ in a range of $O(N)$ values would need
  to pair with a composite $q = N - p$---an event whose probability decreases
  exponentially with $N$ under any reasonable independence model.  The
  non-emptiness is therefore not a statistical coincidence but a consequence of
  the overwhelming prime density within the conductor band.
\end{remark}

\subsection{Bandwidth constancy}

\begin{proposition}[Bandwidth invariance]\label{prop:bw}
  The Goldbach bandwidth $\mathrm{BW}_{\mathrm{GB}}(N) =
  \rho_{\max}^{\mathrm{GB}} - \rho_{\min}^{\mathrm{GB}}$ satisfies
  \[
    \mathrm{BW}_{\mathrm{GB}} = 1.09 \pm 0.20
  \]
  across all 513 values of $N$ in $[1024, 2048]$.  Crucially, the bandwidth is
  independent of the conduit contribution $\xi$: whether $\rad_{\mathrm{odd}}(M) = 1$
  or $\rad_{\mathrm{odd}}(M) = 887$, the Goldbach $\rho$-band has approximately the
  same width.
\end{proposition}

\begin{remark}[Dynamic stability]\label{rem:dynamic}
  Propositions~\ref{prop:corr} and~\ref{prop:bw} together establish what we call
  \emph{dynamic stability}: the Goldbach $\rho$-band shifts rigidly with the static
  conduit (via the Band Shifting Law), but its \emph{internal structure}---the
  bandwidth, the clustering, the ground-state floor---remains invariant.
  The band does not stretch, compress, or deform under conductor perturbations;
  it translates as a rigid body in $\rho$-space.

  This is the precise sense in which the Goldbach locus ``adapts'' to its
  conductor environment: the prime pairs reorganise to track the shifting conduit,
  while preserving their structural rigidity.
\end{remark}

\section{Discussion: Why Propagation Fails but Stability Persists}\label{sec:discussion}

The original motivation for this study was to construct a propagation operator
$\mathcal{P}: \mathcal{M}_N \to \mathcal{M}_{N+2}$ that would carry Goldbach pairs
from $N$ to $N + 2$, enabling an inductive proof.  Let us now explain precisely
why this fails, and why the failure is itself the most important finding.

\subsection{The three obstacles to propagation}

\begin{enumerate}
\item \textbf{Conduit discontinuity.}  The static conduit $\rad_{\mathrm{odd}}(M)$
  can jump from $1$ to $57$ in a single step ($N = 1024 \to 1026$), producing
  $\Delta\rho > 2$.  No continuous operator can model this.

\item \textbf{Ground state identity change.}  The ground-state Goldbach pair
  $(p_0, q_0)$ at $N$ has no algebraic relationship to the ground state at $N + 2$.
  For example, $(3, 1021)$ is the ground state at $N = 1024$, but $1023 = 3 \times 341$
  is composite, so this pair does not ``propagate'' to $N = 1026$.

\item \textbf{Band position depends on $N$.}  The Goldbach $\rho$-band sits at
  $\langle\rho\rangle \approx 3.4$ when $\xi = 0$ (algebraic vacuum) but at
  $\langle\rho\rangle \approx 7.2$ when $\xi \approx 1.9$ (large conduit).
  There is no fixed ``stability band'' that the induction could preserve.
\end{enumerate}

\subsection{What replaces propagation: the adaptive ground state}

The resolution of these obstacles is the central conceptual contribution of this
paper.  We observe that although no \emph{individual} Goldbach pair propagates from
$N$ to $N + 2$, the Goldbach \emph{locus as a collective} exhibits a remarkable
adaptive property:

\begin{enumerate}
\item The conduit jumps are large but \emph{deterministic}: the Band Shifting Law
  predicts $99.7\%$ of the variance in $\langle\rho\rangle$.

\item The band \emph{width} is invariant ($1.09 \pm 0.20$): the locus does not
  stretch or collapse under conduit perturbations.

\item At every $N$, the prime number theorem guarantees a supply of primes in
  $[1, N]$.  The Goldbach locus reorganises by selecting new pairs $(p', q')$ that
  are adapted to the current conduit environment, landing in the shifted band at
  position $\alpha\,\xi(N) + \gamma$.
\end{enumerate}

This is \emph{dynamic stability}: not the stability of individual orbits, but
the stability of the \emph{collective structure} of the locus under arbitrary
perturbations of the conductor environment.

\begin{remark}[Automorphic perspective on non-emptiness]\label{rem:ST}
  From the viewpoint of the $\GSp(4)$ family, the bandwidth constancy has a
  deeper interpretation.  The Sato--Tate equidistribution conjecture for
  genus~$2$ Jacobians predicts that the Frobenius characteristic angles of
  $\Jac(C_{N,p})$ become uniformly distributed as $N \to \infty$.  For this
  equidistribution to hold, the family must contain representatives across
  all conductor levels within the band---in particular, it must contain members
  at the ground state ($\rho_{\min}$), which can only be supplied by Goldbach
  pairs.  The non-emptiness of $\mathscr{G}(N)$ is thus consistent with
  (though not proved by) the requirement that the $\GSp(4)$ family be
  sufficiently ``spectrally complete'' to support Sato--Tate statistics.
  More concretely, the bandwidth invariance ($1.09 \pm 0.20$) ensures that the
  support of the Goldbach locus in the moduli space remains macroscopically thick,
  preventing the Sato--Tate trace distribution from collapsing into degenerate
  sub-locus signatures.
  We emphasise that this is an interpretive remark, not a proof: the logical
  direction from Sato--Tate to Goldbach non-emptiness remains an open problem.
\end{remark}

\begin{remark}[Open question: a measure-theoretic transfer operator]\label{rem:open}
  The failure of a pointwise propagation operator $\mathcal{P}$ raises a natural
  successor question: is it possible to define a modified,
  arithmetic-geometric transfer operator that acts not on individual pairs,
  but on the \emph{measure} of the locus $\mathscr{G}(N)$, absorbing the
  deterministic jumps of the Band Shifting Law?  Such an operator would map
  the Goldbach counting measure at $N$ to a predicted measure at $N + 2$,
  with the Band Shifting Law providing the deterministic drift and the
  residual $3.8\%$ variance supplying a stochastic correction.  This
  perspective connects the conductor rigidity framework to the theory of
  dynamical systems on arithmetic moduli spaces.
\end{remark}

\section{The Analytic Gap}\label{sec:gap}

\begin{remark}[What is and what is not proved]\label{rem:honest}
  \emph{Proved unconditionally:}
  \begin{itemize}
  \item The conduit--boundary decomposition~\eqref{eq:decomp} and the algebraic
    origin of the slope $\alpha \approx 2$.
  \item The conduit discontinuity (Proposition~\ref{prop:jump}).
  \end{itemize}
  \emph{Established computationally ($N \in [1024, 2048]$):}
  \begin{itemize}
  \item The Band Shifting Law with $R^2 = 0.9974$ (Theorem~\ref{thm:BSL}).
  \item The jump--conduit correlation of $0.981$ (Proposition~\ref{prop:corr}).
  \item Bandwidth constancy at $1.09 \pm 0.20$ (Proposition~\ref{prop:bw}).
  \item $100\%$ non-emptiness over $[1024, 2048]$ (Observation~\ref{obs:nonempty}).
  \end{itemize}
  \emph{Not proved:}
  \begin{itemize}
  \item That the Band Shifting Law holds for all $N$ (it is an empirical law in this
    range).
  \item That $\mathscr{G}(N) \neq \emptyset$ for all even $N > 4$.  This is the
    Goldbach conjecture, and the dynamic stability framework, while strongly
    suggestive, does not close the analytic gap identified
    in~\cite{Chen2026GMII}.
  \end{itemize}
\end{remark}

\section{Conclusion}

The conductor orbit propagation from $N = 2^{10}$ to $N = 2^{11}$ reveals that
the Goldbach phenomenon is characterised by \emph{dynamic stability}: the
conductor environment fluctuates wildly (with step-to-step jumps exceeding
$\pm 3$ units in $\rho$), but the Goldbach locus adapts by reorganising its
prime-pair content while preserving three structural invariants---the Band Shifting
Law, the bandwidth constancy, and (computationally) the non-emptiness.

The Band Shifting Law
$\langle\rho\rangle = 1.994\,\xi + 3.424$ ($R^2 = 0.9974$) transforms the
apparent ``chaos'' of conductor evolution into a near-perfect linear relation,
with the slope $\alpha \approx 2$ directly traceable to the algebraic structure
of the discriminant.  The $99.7\%$ of explained variance, combined with the
$98.1\%$ jump--conduit correlation, demonstrate that the conductor geometry of
the Goldbach family is governed by a simple, deterministic law modulated by
small arithmetic fluctuations.

The corrected form $\langle\rho\rangle = 2\,\xi + 4 - c/\log N$ (Remark~\ref{rem:drift})
shows that the Band Shifting Law possesses \emph{asymptotic scale invariance}:
since both the conduit contribution $\xi$ and the intercept correction are
normalised by $\log N$, the law's structure is preserved under rescaling
$N \to \lambda N$.  The local $R^2 \geq 0.997$ within any dyadic interval
$[2^k, 2^{k+1}]$, and its monotonic improvement with $k$ (reaching $0.9994$
at $[2^{13}, 2^{14}]$), suggest that the Band Shifting Law becomes
\emph{asymptotically exact} as $N \to \infty$.
